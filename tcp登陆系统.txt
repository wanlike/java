import java.io.*;
import java.net.*;
public class Logserver {
    public static void main(String[] args)throws Exception{
        ServerSocket ss = new ServerSocket(10005);
        while(true){
            Socket s = ss.accept();
            new Thread(new Usertext(s)).start();
        }
    }
}
